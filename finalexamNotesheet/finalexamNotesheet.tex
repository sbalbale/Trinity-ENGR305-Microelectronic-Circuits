\section*{1. Semiconductor Physics (Ch. 3)}
\textbf{Obj 1-4, 7 | HW 1}

\textbf{Intrinsic Carrier Concentration ($n_i$)}
\begin{itemize}
    \item $n_i = B T^{3/2} e^{-E_g / 2kT}$
    \item \textbf{Si at 300K:} $n_i \approx 1.5 \times 10^{10} \text{ cm}^{-3}$
    \item $k = 8.62 \times 10^{-5} \text{ eV/K}$
    \item $E_g \text{ (Si)} = 1.12 \text{ eV}$
\end{itemize}

\textbf{Carrier Concentrations (Mass Action Law)}
\begin{itemize}
    \item $n \cdot p = n_i^2$
    \item \textbf{N-type ($N_D \gg n_i$):} 
    $$ n_n \approx N_D, \quad p_n \approx \frac{n_i^2}{N_D} $$
    \item \textbf{P-type ($N_A \gg n_i$):} 
    $$ p_p \approx N_A, \quad n_p \approx \frac{n_i^2}{N_A} $$
\end{itemize}

\textbf{Resistivity ($\rho$) \& Conductivity ($\sigma$)}
\begin{itemize}
    \item $\sigma = q(n\mu_n + p\mu_p)$
    \item $\rho = 1/\sigma$
    \item $R = \rho (L/A)$
\end{itemize}

\textbf{pn Junction Physics}
\begin{itemize}
    \item \textbf{Built-in Potential ($V_{bi}$):}
    $$ V_{bi} = V_T \ln \left( \frac{N_A N_D}{n_i^2} \right) $$
    \item \textbf{Thermal Voltage:} $V_T \approx 25.9 \text{ mV}$ (at 300K)
    \item \textbf{Depletion Width ($W$):}
    $$ W = \sqrt{ \frac{2\epsilon_s}{q} \left( \frac{1}{N_A} + \frac{1}{N_D} \right) (V_{bi} + V_R) } $$
    (Set $V_R=0$ for zero bias). $\epsilon_s = \epsilon_0 \epsilon_r$.
\end{itemize}

\section*{2. Diode Circuits (Ch. 4)}
\textbf{Obj 5-6 | HW 2, HW 3}

\textbf{Exponential Model (Shockley)}
$$ I_D = I_S \left( e^{V_D / nV_T} - 1 \right) $$
\textbf{Voltage Change Rule:}
$$ V_2 - V_1 = nV_T \ln\left(\frac{I_2}{I_1}\right) $$
(Assumption: $e^{V/nV_T} \gg 1$)

\textbf{Circuit Models}
\begin{itemize}
    \item \textbf{Ideal:} Forward $V_D=0$ (Short); Reverse $I_D=0$ (Open).
    \item \textbf{CVD:} Forward $V_D=0.7$V; Reverse $I_D=0$.
\end{itemize}

\section*{3. MOSFET DC Circuits (Ch. 5)}
\textbf{Obj 8-10 | HW 4}

\textbf{NMOS Operation ($V_{tn} > 0$)}
\begin{itemize}
    \item \textbf{Overdrive:} $V_{ov} = V_{GS} - V_{tn}$
    \item \textbf{Triode} ($V_{DS} < V_{ov}$):
    $$ I_D = k_n' \frac{W}{L} \left[ (V_{GS} - V_{tn})V_{DS} - \frac{1}{2}V_{DS}^2 \right] $$
    \item \textbf{Saturation} ($V_{DS} \ge V_{ov}$):
    $$ I_D = \frac{1}{2} k_n' \frac{W}{L} (V_{GS} - V_{tn})^2 (1 + \lambda V_{DS}) $$
    (Include $\lambda$ term for Channel Length Modulation).
\end{itemize}

\textbf{PMOS Operation}
\begin{itemize}
    \item Replace $V_{GS}, V_{tn}, V_{DS}$ with absolute values: $|V_{GS}|, |V_{tp}|, |V_{SD}|$.
    \item Saturation when $V_{SD} \ge |V_{ov}|$.
\end{itemize}

\section*{4. BJT DC Circuits (Ch. 6)}
\textbf{Obj 11-12 | HW 5}

\textbf{Parameter Relationships}
\begin{itemize}
    \item $i_E = i_C + i_B$
    \item $i_C = \beta i_B \quad \text{and} \quad i_C = \alpha i_E$
    \item $\alpha = \frac{\beta}{\beta + 1}, \quad \beta = \frac{\alpha}{1 - \alpha}$
\end{itemize}

\textbf{Active Mode Operation}
\begin{itemize}
    \item Base-Emitter: $V_{BE} \approx 0.7$ V
    \item Current: $I_C = I_S e^{V_{BE}/V_T}$
    \item Condition: $V_{CE} > 0.3$ V (NPN)
\end{itemize}

\section*{5. Small Signal Analysis (Ch. 7)}
\textbf{Obj 13-20 | HW 6-9}

\textbf{MOSFET Parameters}
\begin{itemize}
    \item $g_m = \frac{2I_D}{V_{ov}} = \sqrt{2 k_n' \frac{W}{L} I_D} = k_n' \frac{W}{L} V_{ov}$
    \item $r_o = \frac{|V_A|}{I_D} = \frac{1}{\lambda I_D}$
\end{itemize}

\textbf{BJT Parameters}
\begin{itemize}
    \item $g_m = \frac{I_C}{V_T}$
    \item $r_\pi = \frac{\beta}{g_m} = \frac{V_T}{I_B}$
    \item $r_e = \frac{\alpha}{g_m} \approx \frac{V_T}{I_E}$
    \item $r_o = \frac{|V_A|}{I_C}$
\end{itemize}

\textbf{Amplifier Configurations}
\begin{itemize}
    \item \textbf{Common-Source (CS):}
    \begin{itemize}
        \item $R_{in} = R_G$ (or $\infty$ if no bias resistors)
        \item $R_o = R_D || r_o$
        \item $A_{vo} = -g_m (R_D || r_o)$
        \item $G_v = \frac{R_{in}}{R_{in} + R_{sig}} A_{vo}$
    \end{itemize}
    \item \textbf{Common-Emitter (CE):}
    \begin{itemize}
        \item $R_{in} = R_B || r_\pi$
        \item $R_o = R_C || r_o$
        \item $A_{vo} = -g_m (R_C || r_o)$
        \item $G_v = \frac{R_{in}}{R_{in} + R_{sig}} A_{vo}$
    \end{itemize}
\end{itemize}

\textbf{VTC Active Region (MOS)}
\begin{itemize}
    \item Point A (Cutoff boundary): $V_{in} = V_{tn}, V_{out} = V_{DD}$.
    \item Point B (Triode boundary): $V_{out} = V_{in} - V_{tn}$.
\end{itemize}

\section*{6. Differential Pairs (Ch. 9)}
\textbf{Obj 21-22 | HW 9}

\textbf{MOS Differential Pair}
\begin{itemize}
    \item \textbf{Bias:} $I_{D1} = I_{D2} = I/2$ (at equilibrium).
    \item \textbf{Overdrive:} $V_{ov} = \sqrt{I / (k_n' (W/L))}$.
    \item \textbf{Diff. Gain:} $|A_d| = g_m R_D$.
\end{itemize}

\textbf{BJT Differential Pair}
\begin{itemize}
    \item \textbf{Bias:} $I_{E1} = I_{E2} = I/2$.
    \item \textbf{Diff. Input Res:} $R_{id} = 2r_\pi$.
    \item \textbf{Diff. Gain:} $|A_d| = g_m R_C$.
\end{itemize}
